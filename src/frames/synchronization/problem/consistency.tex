\begin{frame}[fragile]{Cohérence de la machine à états répliquée}

  \on[top,text]{
    \begin{algorithm}[H]
      \LVariables{}{
        $\mathit{state}_i \leftarrow q_0$\;
      }
      \Operation{$\Execute(c \in C)  \in R$}{
        \Let $r\leftarrow \rho(\mathit{state}_i, c)$\;
        \Broadcast{rb} $\textsc{command}(c)$\;
        \Return $r$\;
      }
      \When{\Deliver{rb} $\textsc{command}(c \in C)$ \From $p_j$}{
        $\mathit{state}_i \leftarrow \tau(\mathit{state}_i, c)$\;
      }
    \end{algorithm}
  }
