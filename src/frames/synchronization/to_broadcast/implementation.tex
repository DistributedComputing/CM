\begin{frame}[fragile]{Implémentation de \Broadcast{to}}

  \onBlock[top=-5mm]{Stratégie}{
    \begin{enumerate}
    \item Implémenter une version affaiblie du consensus
      \begin{description}
      \item[Accord: ] Au plus une valeur décidée
      \item[Validité: ] Toute valeur décidée a été proposée
      \item[Terminaison: ] \sout{Tout processus correct finit par décider une valeur}
        \begin{itemize}
        \item Tout processus correct décide \structure{avec probabilité $1$}
          \begin{itemize}
          \item Exemple : Ben-Or
          \end{itemize}
        \item Les processus décident \structure{si le système est assez stable}
          \begin{itemize}
          \item Exemple : Paxos
          \end{itemize}
        \end{itemize}
      \end{description}
    \item\vspace{3mm} Réduction de consensus à \Broadcast{to}
      \begin{itemize}
      \item On suppose qu'on a le consensus, implémenter \Broadcast{to}. 
      \end{itemize}
    \item\vspace{3mm} Réduction de \Broadcast{to} à la machine répliquée
      \begin{itemize}
      \item On suppose qu'on a \Broadcast{to}, implémenter le consensus. 
      \end{itemize}
    \end{enumerate}
  }
