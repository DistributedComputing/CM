\SetKwFunction{Propose}{propose}

\begin{frame}[fragile]{Le consensus}

  \on[text, top]{
    Chaque processus \emph{propose} une valeur et peut en \emph{décider} une.
    \begin{description}
    \item[Accord: ] Au plus une valeur décidée
    \item[Validité: ] Toute valeur décidée a été proposée
    \item[Terminaison: ] Tout processus correct finit par décider une valeur
    \end{description}
  }
  
  \onAlertBlock<2->[y=9mm]{Théorème (FLP85)}{
    Il est impossible de résoudre le consensus dans $\mathcal{CAMP}_{n, t}[t\ge 1]$
  }

  \onAlertBlock<3->[y=-17mm]{Réduction de \Broadcast{to} au consensus}{
    \begin{algorithm}[H]
      \lLVariables{}{$\mathit{decided}_i \leftarrow \bot$}
      \Operation{$\Propose(v)$}{
        \SBroadcast{to} $\textsc{propose}(v)$\;
        \Return $\mathit{decided}_i$\;
      }
      \When{\Deliver{to} $\textsc{propose}(v)$ \From $p_j$}{
        \lIf{$\mathit{decided}_i = \bot$}{$\mathit{decided}_i \leftarrow v$}
      }
    \end{algorithm}
  }

  \on<2->[bottom=-2mm,text]{
    \begin{citing}
    \item[FLP85] Michael J. Fischer, Nancy A. Lynch, Michael S. Paterson. \textit{Impossibility of Distributed Consensus with One Faulty Process.} JACM (1985)
    \end{citing}
  }

\end{frame}
